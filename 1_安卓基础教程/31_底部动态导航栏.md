### 1.创建四个图标文件fill，然后把他们复制一下转成stroke模式。再创建一个ConstraintLayout，转成MotionLayout

```xml
<?xml version="1.0" encoding="utf-8"?>
<androidx.constraintlayout.motion.widget.MotionLayout xmlns:android="http://schemas.android.com/apk/res/android"
    xmlns:app="http://schemas.android.com/apk/res-auto"
    android:id="@+id/groupMotionLayout"
    android:layout_width="match_parent"
    android:layout_height="match_parent"
    app:layoutDescription="@xml/navigation_menu_scene">

    <androidx.constraintlayout.utils.widget.ImageFilterView
        android:id="@+id/imageFilter"
        android:layout_width="wrap_content"
        android:layout_height="wrap_content"
        android:src="@drawable/group_24px_stroke"
        app:altSrc="@drawable/group_24px_fill"
        app:layout_constraintBottom_toTopOf="@id/textView"
        app:layout_constraintEnd_toEndOf="parent"
        app:layout_constraintStart_toStartOf="parent"
        app:layout_constraintTop_toTopOf="parent" />

    <TextView
        android:id="@+id/textView"
        android:layout_width="wrap_content"
        android:layout_height="wrap_content"
        android:text="@string/navigation_group_text"
        app:layout_constraintStart_toStartOf="parent"
        app:layout_constraintEnd_toEndOf="parent"
        app:layout_constraintTop_toBottomOf="@id/imageFilter"
        app:layout_constraintBottom_toBottomOf="parent"
        android:textColor="?attr/colorControlNormal"/>

</androidx.constraintlayout.motion.widget.MotionLayout>
```

### 2.设置动画

```xml
<?xml version="1.0" encoding="utf-8"?>
<MotionScene xmlns:android="http://schemas.android.com/apk/res/android"
    xmlns:motion="http://schemas.android.com/apk/res-auto">

    <Transition
        motion:constraintSetStart="@+id/start"
        motion:constraintSetEnd="@+id/end"
        motion:duration="200">
        <KeyFrameSet>

            <KeyAttribute
                motion:motionTarget="@id/imageFilter"
                motion:framePosition="15"
                android:translationX="-6dp" />

            <KeyAttribute
                motion:motionTarget="@id/imageFilter"
                motion:framePosition="30"
                android:translationX="6dp" />

            <KeyAttribute
                motion:motionTarget="@id/imageFilter"
                motion:framePosition="45"
                android:translationX="-4dp" />

            <KeyAttribute
                motion:motionTarget="@id/imageFilter"
                motion:framePosition="60"
                android:translationX="4dp" />

            <KeyAttribute
                motion:motionTarget="@id/imageFilter"
                motion:framePosition="75"
                android:translationX="0dp" />

        </KeyFrameSet>

    </Transition>

    <ConstraintSet android:id="@+id/start">

        <Constraint android:id="@id/imageFilter">

            <Transform
                android:translationX="0dp" />

            <CustomAttribute
                motion:attributeName="crossfade"
                motion:customFloatValue="0" />

        </Constraint>

        <Constraint android:id="@id/textView">

            <CustomAttribute
                motion:attributeName="textColor"
                motion:customColorValue="?attr/colorControlNormal" />

        </Constraint>

    </ConstraintSet>

    <ConstraintSet android:id="@+id/end">

        <Constraint android:id="@id/imageFilter">

            <Transform
                android:translationX="0dp" />

            <CustomAttribute
                motion:attributeName="crossfade"
                motion:customFloatValue="1" />

        </Constraint>

        <Constraint android:id="@id/textView">

            <CustomAttribute
                motion:attributeName="textColor"
                motion:customColorValue="?attr/colorPrimary" />

        </Constraint>

    </ConstraintSet>

</MotionScene>
```

### 4.最后在代码里驱动一下，加了切换的动画

```kotlin
package com.example.motionapplication

import android.os.Bundle
import androidx.activity.enableEdgeToEdge
import androidx.appcompat.app.AppCompatActivity
import androidx.constraintlayout.motion.widget.MotionLayout
import androidx.core.view.ViewCompat
import androidx.core.view.WindowInsetsCompat
import androidx.navigation.NavController
import androidx.navigation.NavOptions
import androidx.navigation.fragment.NavHostFragment

class MainActivity : AppCompatActivity() {

    private lateinit var navHostFragment: NavHostFragment

    private lateinit var messageMotionLayout: MotionLayout
    private lateinit var groupMotionLayout: MotionLayout
    private lateinit var exploreMotionLayout: MotionLayout
    private lateinit var profileMotionLayout: MotionLayout

    private var currentIndex = 0

    override fun onCreate(savedInstanceState: Bundle?) {
        super.onCreate(savedInstanceState)
        enableEdgeToEdge()
        setContentView(R.layout.activity_main)
        ViewCompat.setOnApplyWindowInsetsListener(findViewById(R.id.main)) { v, insets ->
            val systemBars = insets.getInsets(WindowInsetsCompat.Type.systemBars())
            v.setPadding(systemBars.left, systemBars.top, systemBars.right, systemBars.bottom)
            insets
        }

        messageMotionLayout = findViewById<MotionLayout>(R.id.messageMotionLayout)
        groupMotionLayout = findViewById<MotionLayout>(R.id.groupMotionLayout)
        exploreMotionLayout = findViewById<MotionLayout>(R.id.exploreMotionLayout)
        profileMotionLayout = findViewById<MotionLayout>(R.id.profileMotionLayout)

        val destinationMap = mapOf<Int, MotionLayout>(
            R.id.messageFragment to messageMotionLayout,
            R.id.groupFragment to groupMotionLayout,
            R.id.exploreFragment to exploreMotionLayout,
            R.id.profileFragment to profileMotionLayout
        )

        navHostFragment =
            supportFragmentManager.findFragmentById(R.id.fragmentContainerView) as NavHostFragment
        val navController: NavController = navHostFragment.navController

        destinationMap.forEach { i, layout ->
            layout.setOnClickListener {
                if(i==navController.currentDestination?.id){
                    return@setOnClickListener
                }else{
                    val targetIndex = destinationMap.keys.indexOf(i)
                    val options =if(currentIndex<targetIndex){
                        NavOptions.Builder()
                            .setEnterAnim(R.anim.slide_in_right)
                            .setExitAnim(R.anim.slide_out_left)
                            .build()
                    }else{
                        NavOptions.Builder()
                            .setEnterAnim(R.anim.slide_in_left)
                            .setExitAnim(R.anim.slide_out_right)
                            .build()
                    }
                    currentIndex = targetIndex
                    navController.navigate(i, null, options)
                }
            }
        }

        navController.addOnDestinationChangedListener { controller, destination, arguments ->
            controller.popBackStack()
            destinationMap.values.forEach { layout -> layout.progress = 0f }
            destinationMap[destination.id]?.transitionToEnd()
        }

    }
}
```
