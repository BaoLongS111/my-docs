### 1.在Entity类新增字段

新增完字段后一定要写getter和setter，不然room框架会报错。

注意如果加了column_info，名字一定要和SQL语句里的字段名字对的上。

### 2.在Database类里写上Migration方法，然后在build前调用addMigration

```
fallbackToDestructiveMigration()
```

这个方法被弃用了，这是个摧毁式的迁移，直接删除所有数据重新创建数据库。

```java Database.java
package com.example.helloworld.Database;

import android.content.Context;

import androidx.annotation.NonNull;
import androidx.room.Database;
import androidx.room.Room;
import androidx.room.RoomDatabase;
import androidx.room.migration.Migration;
import androidx.sqlite.db.SupportSQLiteDatabase;

import com.example.helloworld.DAO.WordDAO;
import com.example.helloworld.Entity.Word;

@Database(entities = {Word.class}, version = 2, exportSchema = false)
public abstract class WordDatabase extends RoomDatabase {
    private static volatile WordDatabase INSTANCE;

    public static WordDatabase getDatabase(Context context) {
        if (INSTANCE == null) {
            synchronized (WordDatabase.class) {
                if (INSTANCE == null)
                {
                    INSTANCE = Room.databaseBuilder(context.getApplicationContext(), WordDatabase.class, "word_database")
                            .addMigrations(MIGRATION_1_2)
                            .build();
                }
            }
        }
        return INSTANCE;
    }

    public abstract WordDAO getWordDAO();

    static final Migration MIGRATION_1_2 = new Migration(1,2) {
        @Override
        public void migrate(@NonNull SupportSQLiteDatabase db) {
            db.execSQL("ALTER TABLE Word ADD COLUMN isChecked INTEGER NOT NULL DEFAULT 0");
        }
    };
}
```

### 3.再把flag变量删除

```java WordDatabase.java
package com.example.helloworld.Database;

import android.content.Context;

import androidx.annotation.NonNull;
import androidx.room.Database;
import androidx.room.Room;
import androidx.room.RoomDatabase;
import androidx.room.migration.Migration;
import androidx.sqlite.db.SupportSQLiteDatabase;

import com.example.helloworld.DAO.WordDAO;
import com.example.helloworld.Entity.Word;

@Database(entities = {Word.class}, version = 3, exportSchema = false)
public abstract class WordDatabase extends RoomDatabase {
    private static volatile WordDatabase INSTANCE;

    public static WordDatabase getDatabase(Context context) {
        if (INSTANCE == null) {
            synchronized (WordDatabase.class) {
                if (INSTANCE == null)
                {
                    INSTANCE = Room.databaseBuilder(context.getApplicationContext(), WordDatabase.class, "word_database")
                            .addMigrations(MIGRATION_2_3)
                            .build();
                }
            }
        }
        return INSTANCE;
    }

    public abstract WordDAO getWordDAO();

    static final Migration MIGRATION_1_2 = new Migration(1,2) {
        @Override
        public void migrate(@NonNull SupportSQLiteDatabase db) {
            db.execSQL("ALTER TABLE Word ADD COLUMN isChecked INTEGER NOT NULL DEFAULT 0");
        }
    };

    static final Migration MIGRATION_2_3 = new Migration(2,3){
        @Override
        public void migrate(@NonNull SupportSQLiteDatabase db) {
            db.execSQL("CREATE TABLE word_temp(id INTEGER PRIMARY KEY NOT NULL,word_name TEXT,chinese_meaning TEXT)");
            db.execSQL("INSERT INTO word_temp(id,word_name,chinese_meaning) "+
                    "SELECT id,word_name,chinese_meaning FROM word");
            db.execSQL("DROP TABLE word");
            db.execSQL("ALTER TABLE word_temp RENAME TO word");
        }
    };
}
```
