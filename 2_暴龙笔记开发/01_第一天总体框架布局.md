### 2.统一设置字体颜色、字体大小、字体文字以及沉浸式状态栏设置

themes.xml

```xml
<resources xmlns:tools="http://schemas.android.com/tools">
    <!-- Base application theme. -->
    <style name="Base.Theme.BaLonNote" parent="Theme.Material3.DayNight.NoActionBar">
        <!-- 全局窗口背景 → 纯白 -->
        <item name="android:windowBackground">@color/main_background_color</item>
        <!-- 全局默认文字颜色 → 深灰，护眼不刺眼 -->
        <item name="android:textColorPrimary">@color/black</item>
        <item name="android:textColorSecondary">@color/text_sub</item>

        <!-- 状态栏基础配置 -->
        <item name="android:statusBarColor">@color/main_background_color</item>
        <item name="android:windowLightStatusBar">true</item>
    </style>

    <style name="Theme.BaLonNote" parent="Base.Theme.BaLonNote" />
</resources>
```

colors.xml

```xml
<?xml version="1.0" encoding="utf-8"?>
<resources>
    <color name="black">#FF000000</color>
    <color name="white">#FFFFFFFF</color>
    <color name="main_background_color">#f7f7f7</color>
    <!-- 次要文字：时间、预览、提示字 -->
    <color name="text_sub">#969696</color>
    <color name="nav_background_color">#fdfdfd</color>
</resources>
```

dimens.xml

```xml
<?xml version="1.0" encoding="utf-8"?>
<resources>
    <dimen name="text_normal_size">12sp</dimen>
</resources>
```
